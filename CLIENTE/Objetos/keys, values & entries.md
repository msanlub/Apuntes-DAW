

```
"use strict"
const producto = {
	name: "Teclado",
	price: 20,
	available: true,
}

console.log(producto)

//mostrar las claves 
console.log(Object.keys(producto)) -> name,price,available

//mostrar valores
console.log(Object.values(producto)) -> teclado,20,true

//ambos
console.log(Object.entries(producto)) -> muestra array clave valor
```
